# Структрура приложения

```plaintext
battle_ship/
│
├── .venv/                      # Виртуальное окружение
│
├── main.py                     # Точка входа, запуск приложения
│
├── requirements.txt            # Список зависимостей
│
├── config.py                   # Константы: размер поля, количество кораблей, цвета
│
├── models/                     # Модели данных и логика
│   ├── __init__.py
│   ├── ship.py                 # Класс корабля (размер, координаты, ориентация)
│   ├── grid.py                 # Сетка игрового поля (клетки, состояния)
│   ├── game.py                 # Основная игровая логика (ходы, проверка победителя)
│   └── player.py               # Игрок и компьютер (возможности ходов)
│
├── views/                      # Отображение и отрисовка
│   ├── __init__.py
│   ├── main_window.py          # Главное окно приложения
│   ├── board_view.py           # Отрисовка одного игрового поля
│   ├── ship_placer.py          # Интерфейс для расстановки кораблей
│   └── ui_elements.py          # Кнопки, статус-бар, панели
│
├── controllers/                # Связь между моделями и отображением
│   ├── __init__.py
│   ├── game_controller.py      # Обработка кликов, управление ходами
│   ├── placement_controller.py # Расстановка кораблей пользователем
│   └── ai_controller.py        # Логика ходов компьютера
│
├── utils/                      # Вспомогательные модули
│   ├── __init__.py
│   ├── coordinates.py          # Преобразование координат (клик → клетка)
│   ├── validators.py           # Проверка расстановки кораблей
│   └── ai_strategies.py        # Разные стратегии AI (случайная, умная)
│
├── assets/                     # Ресурсы (опционально)
│   ├── images/
│   │   ├── ship.png
│   │   ├── hit.png
│   │   └── miss.png
│   └── sounds/
│       └── shot.wav
│
└── tests/                      # Модульные тесты
    ├── __init__.py
    ├── test_grid.py
    ├── test_ship.py
    └── test_game.py
```
